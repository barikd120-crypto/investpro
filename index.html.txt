<script>
let wallet=1000, invested=0;
function login(){
const u=document.getElementById('username').value;
if(!u)return alert('Enter username');
document.getElementById('auth').classList.add('hidden');
document.getElementById('dashboard').classList.remove('hidden');
document.getElementById('userLabel').innerText=u;
document.getElementById('refCode').innerText='REF-'+Math.floor(Math.random()*100000);
}
function logout(){location.reload()}
function show(id){
['dash','invest','referral','withdraw','admin'].forEach(s=>document.getElementById(s).classList.add('hidden'));
document.getElementById(id).classList.remove('hidden');
}
function invest(){
const amt=Number(document.getElementById('amount').value);
if(amt>wallet)return alert('Insufficient balance');
wallet-=amt; invested+=amt;
document.getElementById('wallet').innerText=wallet;
document.getElementById('invested').innerText=invested;
document.getElementById('profit').innerText=Math.floor(invested*0.02);
alert('Investment successful (demo)');
}
